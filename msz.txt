protected bool Validate()
    {
        div_error.Visible = false;
        div_error1.Visible = false;
        if (txtfname.Text.Trim() == "")
        {
            Message = "Sorry, " + MSG.Show(MsgType.Valid_FName);
            txtfname.Focus();
            SetErrror(Message);
            return false;
        }
        if (ddlgender.SelectedIndex == 0)
        {
            Message = "Sorry, " + MSG.Show(MsgType.Select_Gender);
            SetErrror(Message);
            return false;
        }
        if (txtdob.Text.Trim() == string.Empty)
        {
            Message = "Sorry, " + MSG.Show(MsgType.Valid_DOB);
            SetErrror(Message);
            return false;
        }
        else
        {
            string[] s = txtdob.Text.Trim().Split('/');
            dobdt = s[1] + "/" + s[0] + "/" + s[2];
            if (Convert.ToInt32(s[2]) < 1900 || Convert.ToInt32(s[2]) > 2015)
            {
                Message = "Sorry Invalid Date of Birth";
                SetErrror(Message);
                return false;
            }
            if (!DateTime.TryParse(dobdt, out Test))
            {
                Message = "Sorry Invalid, " + MSG.Show(MsgType.Valid_DOB);
                SetErrror(Message);
                return false;
            }
        }
        if (txtmobile.Text.Trim() == string.Empty)
        {
            Message = "Sorry, " + MSG.Show(MsgType.Fill_MobNo);
            SetErrror(Message);
            return false;
        }
        if (txtpincode.Text.Trim() == string.Empty)
        {
            Message = "Sorry, " + MSG.Show(MsgType.Fill_Pincode);
            SetErrror(Message);
            return false;
        }
        if (txtemail.Text.Trim() == string.Empty)
        {
            Message = "Sorry, " + MSG.Show(MsgType.Required_emailid);
            SetErrror(Message);
            return false;
        }
      
        if (objDUT.IsExist("Select Count(Loginid) from Member_Master where emailid='" + txtemail.Text.Trim() + "'") == true)
        {
            Message = "Sorry, " + MSG.Show(MsgType.EmailIDExist);
            SetErrror(Message);
            return false;
        }
        if (txtcardno.Text.Trim() == string.Empty)
        {
            Message = "Sorry," + MSG.Show(MsgType.CardNo);
            SetErrror(Message);
            return false;
        }
        if (txtcardno.Text.Trim().IndexOf(' ') > -1)
        {
            Message = "Sorry, " + MSG.Show(MsgType.CardNoSpace);
            SetErrror(Message);
            return false;
        }
        if (txtrecardno.Text.Trim() == string.Empty)
        {
            Message = "Sorry," + MSG.Show(MsgType.CardNo);
            SetErrror(Message);
            return false;
        }
        if (txtrecardno.Text.Trim().IndexOf(' ') > -1)
        {
            Message = "Sorry, " + MSG.Show(MsgType.CardNoSpace);
            SetErrror(Message);
            return false;
        }
        int Pcnt = Convert.ToInt32(objDUT.GetScalar("Select count(*) from Epin_master where EpinNumber='" + txtcardno.Text.Trim() + "' and regNo=0 AND sepinid<>0 and eactive=1 "));
        if (Pcnt == 0)
        {
            Message = "Sorry, " + MSG.Show(MsgType.NotExist_Evoucher);
            SetErrror(Message);
            return false;
        }
        if (txtpassword.Text.Trim() == string.Empty)
        {
            Message = "Sorry," + MSG.Show(MsgType.Fill_Password);
            SetErrror(Message);
            return false;
        }
        if (txtpassword.Text.Trim().Length < 4)
        {
            Message = "Sorry, " + MSG.Show(MsgType.Min_Password);
            SetErrror(Message);
            return false;
        }
        if (txtpassword.Text.Trim().IndexOf(' ') > -1)
        {
            Message = "Sorry, " + MSG.Show(MsgType.spacenotallowpwd);
            SetErrror(Message);
            return false;
        }
       
        div_error.Visible = false;
        return true;

    }

    [WebMethod]
    public List<MemberMail> GetSentMails_Member(Message_Param message)
    {
        List<MemberMail> mailList = new List<MemberMail>();
        try
        {
            SqlParameter[] param = new SqlParameter[4];
            param[0] = new SqlParameter("@PageIndex", SqlDbType.Int);
            param[0].Value = message.PageIndex;
            param[1] = new SqlParameter("@PageSize", SqlDbType.Int);
            param[1].Value = message.PageSize;
            param[2] = new SqlParameter("@loginid", SqlDbType.VarChar, 500);
            param[2].Value = message.Loginid;
            param[3] = new SqlParameter("@regno", SqlDbType.BigInt);
            param[3].Value = message.Regno;
            sdr = objDUT.GetDataReaderSP(param, "SP_SentMemberMailbox");
            while (sdr.Read())
            {
                MemberMail mail = new MemberMail();
                mail.msgID = Convert.ToInt64(sdr["MsgID"].ToString());
                mail.msgType = sdr["MsgType"].ToString();
                mail.msgSubject = sdr["MsgSubject"].ToString();
                mail.fromRegNo = sdr["LoginID"].ToString();
                mail.toRegNo = sdr["LoginID"].ToString();
                mail.msgDate = sdr["Date"].ToString();
                mail.Message = sdr["Message"].ToString();
                mail.smalltext = sdr["Smalltext"].ToString();
                mail.seen = sdr["msgseen"].ToString();
                mail.Regno = sdr["FromRegNo"].ToString();
                mail.Attachments = sdr["MsgAttachments"].ToString();
                mailList.Add(mail);
            }
        }
        catch (Exception ex)
        {


        }
        return mailList;
    }
