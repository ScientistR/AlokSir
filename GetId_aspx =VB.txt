            $(".autosuggest").autocomplete({

                    $.ajax({
                        type: "POST",

                        url: "getdata.asmx/GetAutoCompleteData",
                        data: "{'username':'" + document.getElementById('txtSearch').value + "'}",
                        dataType: "json",

                            response(data.d);
                        },
                        error: function (result) {

                    });

            });
        }
